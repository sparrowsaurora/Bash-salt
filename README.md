## Added BASH Commands

    "la" = ls -a
    "rf" = rm -rf
    "c" = clear
    "b" = cd ..
    "gl" = git log --oneline --decorate --all
    "git-a" = git add . && git commit -m $
    "mkcd" = mkdir $ && cd $
    "view" = cd $ && la && b
    "todo" = todo
    "wordle" = a wordle game
    "tree" = prints a file tree
